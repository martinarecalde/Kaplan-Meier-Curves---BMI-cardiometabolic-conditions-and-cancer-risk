Kaplan Meier curves

library(survival)
library(survminer)
library(dplyr)
library(ranger)
library(ggplot2)
library(ggfortify)
data_total<- readRDS("//EPOFS/RWCancerEpi/COMOCANS-AdCoCans-OBECAN/Martina/Adcocans/data_total.rds")

#Kaplan Meier curves

BMI

survfit.bmi <- survfit(Surv(age_index, age_index + time, status) ~ bmi.gr, data = data_total)
autoplot(survfit.bmi, censor = FALSE)

Cardiometabolic Conditions

survfit.cardiometa<-survfit(Surv(age_index, age_index + time, status) ~ CurrentState, 
                        data = data_total)
autoplot(survfit.cardiometa, censor = FALSE)

Age

data_total<-data_total%>%mutate(age.gr= ifelse(age_index<=50, "Aged 40 to 50",
                                                           ifelse(age_index<=60, "Aged 51 to 60",
                                                                  ifelse(age_index<=70, "Aged 61 to 70",
                                                                         "Aged 80 or above" ))))

survfit.age<-survfit(Surv(age_index, age_index + time, status) ~ age.gr, 
                        data = data_total)
autoplot(survfit.age, censor = FALSE)

Sex

survfit.sex<-survfit(Surv(age_index, age_index + time, status) ~ sex, 
                        data = data_total)
autoplot(survfit.sex, censor = FALSE)

MEDEA

survfit.medea<-survfit(Surv(age_index, age_index + time, status) ~ qmedea, 
                        data = data_total)
autoplot(survfit.medea, censor = FALSE)

Nationality
